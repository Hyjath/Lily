Option Explicit


Public Sub setProbability(ByVal nProbablity As Integer)
    On Error GoTo ErrorHandler
    Dim oInspector As Lime.Inspector
    
    Set oInspector = Application.ActiveInspector
    
    If Globals.VerifyInspector("business", oInspector, False) Then
        If nProbablity >= 0 And nProbablity <= 100 Then
            Call oInspector.Controls.SetValue("probability", nProbablity / 100)
        End If
    
    End If
    
    Exit Sub
ErrorHandler:
    Call UI.ShowError("Actionpad_Business.SetProbability")
End Sub
